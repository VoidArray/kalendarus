log-level = "info"
calendar-url = "http://127.0.0.1/basic.ics"
timezone = "Asia/Yekaterinburg"

    token = "bot123456:ABC-DEF1234ghIkl-zyx57W2v1u123ew11"
    chat-id = "12345678"

[plainfile]

    data_dir = "/etc/kalendarus/data"
